% Настройки заголовков (titlesec)
\usepackage{titlesec}

% Настройка заголовков структурных элементов (CHAPTER* без номера)
% Согласно п.2.6.2 - посередине строки, прописными буквами, без выделения шрифта
\titleformat{name=\chapter,numberless}[block]
{\normalfont\normalsize\centering\MakeUppercase}  % Обычный шрифт, прописные буквы, по центру
{}                                                % Без номера
{0pt}                                            % Без отступа между номером и названием
{}                                               % Дополнительный код (пустой)

\titlespacing*{name=\chapter,numberless}{0pt}{15pt}{15pt} % Отступы согласно п.2.7

% Настройка заголовков пронумерованных глав (CHAPTER)
% Согласно п.2.6.1 - с абзацного отступа, с прописной буквы, полужирным шрифтом
\titleformat{\chapter}[block]
{\normalfont\normalsize\bfseries}               % Обычный размер, полужирный шрифт
{\thechapter}                                   % Номер главы  
{1em}                                          % Отступ между номером и названием
{}                                             % Дополнительный код (пустой)

\titlespacing*{\chapter}{1.25cm}{15pt}{15pt}   % Отступы: абзацный отступ, 15pt сверху и снизу

% Настройка заголовков SECTION
% Согласно п.2.6.1 - с абзацного отступа, с прописной буквы, полужирным шрифтом  
\titleformat{\section}[block]
{\normalfont\normalsize\bfseries}               % Обычный размер, полужирный шрифт
{\thesection}                                   % Номер секции
{1em}                                          % Отступ между номером и названием  
{}                                             % Дополнительный код (пустой)

\titlespacing*{\section}{1.25cm}{15pt}{15pt}   % Отступы: абзацный отступ, 15pt сверху и снизу

% Настройка заголовков SUBSECTION
% Согласно п.2.6.1 - с абзацного отступа, с прописной буквы, полужирным шрифтом
\titleformat{\subsection}[block]
{\normalfont\normalsize\bfseries}               % Обычный размер, полужирный шрифт
{\thesubsection}                                % Номер подсекции
{1em}                                          % Отступ между номером и названием
{}                                             % Дополнительный код (пустой)

\titlespacing*{\subsection}{1.25cm}{15pt}{15pt} % Отступы: абзацный отступ, 15pt сверху и снизу
